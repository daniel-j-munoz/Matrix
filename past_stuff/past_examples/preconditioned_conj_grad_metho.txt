Matrix Solve::preconditioned_conjugate_gradient(Matrix A, Matrix x, Matrix M){
    float e = 1e-6;
    Matrix b = A.get_column(A.N - 1);
    A.resize(A.M, A.N - 1);

    // Residual
    Matrix pred = A * x; 
    Matrix r = b - pred;
    
    // z = M^(-1) * r  (apply preconditioner)
    Matrix z = M.solve(r);  // or M_inverse * r
    
    // P
    Matrix p = z;  // Initialize with preconditioned residual
    
    Matrix past_z(0, 0);
    Matrix past_r(0, 0);

    for(int i = 0; i < 100; i++){
        // Update x (same alpha formula)
        float alpha = r.dot(z) / (p.T() * A * p).get(0, 0);  // r^T * z instead of r^T * r
        Matrix v = p.copy();
        v.scale(alpha);
        x = x + v; 

        // Update residual 
        pred = A * x; 
        past_r = r;
        past_z = z;
        r = b - pred;

        if(r.magnitude() < e){
            cout << "converged\n";
            break;
        } else {
            // Apply preconditioner to new residual
            z = M.solve(r);  // z = M^(-1) * r
            
            // Update P
            float beta = r.dot(z) / past_r.dot(past_z);  // r^T * z instead of r^T * r
            p.scale(beta); 
            p = p + z;  // Add z instead of r
        }
    }

    return x;
}
